JavaScript Implementation
// Enhanced Download Handler
function handleDownload(button, filename) {
  // Add downloading state
  button.classList.add('downloads-downloading');
  button.innerHTML = `
    <svg class="animate-spin downloads-icon" viewBox="0 0 24 24">
      <circle cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4" fill="none"></circle>
      <path d="M4 12a8 8 0 0 1 8-8v4a4 4 0 0 0-4 4z" fill="currentColor"></path>
    </svg>
  `;
  
  // Simulate download delay
  setTimeout(() => {
    // Change to downloaded state
    button.classList.remove('downloads-downloading');
    button.classList.add('downloads-downloaded');
    button.innerHTML = `
      <svg class="downloads-icon" viewBox="0 0 24 24">
        <path d="M20 6L9 17l-5-5" stroke="white" stroke-width="2" fill="none"></path>
      </svg>
    `;
    
    // Reset after 2 seconds
    setTimeout(() => {
      button.classList.remove('downloads-downloaded');
      button.innerHTML = `
        <svg class="downloads-icon download-icon-svg" viewBox="0 0 24 24">
          <path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4"></path>
          <polyline points="7,10 12,15 17,10"></polyline>
          <line x1="12" y1="15" x2="12" y2="3"></line>
        </svg>
      `;
    }, 2000);
  }, 1500);
  
  // Actual download logic would go here
  console.log('Downloading:', filename);
}
