JavaScript Implementation
// Enhanced News List with Dynamic Loading
class NewsListComponent {
  constructor(container) {
    this.container = container;
    this.newsItems = [];
    this.init();
  }
  
  init() {
    // Intersection Observer for animations
    this.observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          entry.target.classList.add('news-item-animate');
          this.observer.unobserve(entry.target);
        }
      });
    }, { threshold: 0.3 });
    
    // Observe all news items
    this.observeNewsItems();
  }
  
  observeNewsItems() {
    const newsItems = this.container.querySelectorAll('.news-item-interactive');
    newsItems.forEach(item => {
      this.observer.observe(item);
    });
  }
  
  // Add new news item dynamically
  addNewsItem(date, category, headline) {
    const newsItemsGroup = this.container.querySelector('.news-items-group');
    
    // Create divider if not first item
    if (newsItemsGroup.children.length > 0) {
      const divider = document.createElement('div');
      divider.className = 'news-divider';
      newsItemsGroup.appendChild(divider);
    }
    
    // Create news item
    const newsItem = document.createElement('div');
    newsItem.className = 'press-release-xl news-item-interactive';
    newsItem.innerHTML = `
      <div class="press-release-text-group">
        <div class="press-release-date">${date}</div>
        <div class="press-release-line"></div>
        <div class="press-release-category">${category}</div>
      </div>
      <div class="press-release-headline">${headline}</div>
    `;
    
    newsItemsGroup.appendChild(newsItem);
    this.observer.observe(newsItem);
  }
  
  // Load more news items
  loadMore() {
    // Simulate API call
    return new Promise(resolve => {
      setTimeout(() => {
        const moreNews = [
          { date: '10.5.2025', category: 'Innovation', headline: 'Neue digitale Services für Fahrgäste' },
          { date: '08.5.2025', category: 'Expansion', headline: 'Netinera erweitert Streckennetz in Norddeutschland' }
        ];
        
        moreNews.forEach(news => {
          this.addNewsItem(news.date, news.category, news.headline);
        });
        
        resolve(moreNews);
      }, 1000);
    });
  }
}

// Initialize component
document.addEventListener('DOMContentLoaded', () => {
  const newsContainers = document.querySelectorAll('.news-layout-group');
  newsContainers.forEach(container => {
    new NewsListComponent(container);
  });
});
